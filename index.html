D3

d3.csv("data/data.csv", function(data){
    d3.select("#chart1")
         .append("g")
         .selectAll(".dot")
         .data(data)
         .enter()
        .append("circle")
        .attr("class","dot")
        .attr("cx",function(d){
             return d.income/30
         })
        .attr("cy", function(d){
             return 600-d.lifeExpectancy*4
        })
         .attr("r", 2)
})
d3.select("#chart1")
.on("click", function(){
     d3.selectAll("#chart1  .dot")
         .transition()
         .attr("cx",function(d){
            return (d.year-1900)*7
        })
})
